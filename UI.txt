ui/main_window.py
from PyQt6.QtWidgets import QMainWindow, QWidget, QHBoxLayout, QVBoxLayout
from ui.sidebar import Sidebar
from ui.topbar import TopBar
from ui.device_panel import DevicePanel
from ui.bottom_tools import BottomTools
class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("eTool - Mobile Unlock Tool")
        self.resize(1200, 800)
        main_widget = QWidget()
        self.setCentralWidget(main_widget)
        main_layout = QVBoxLayout(main_widget)
        main_layout.setContentsMargins(0, 0, 0, 0)
        # Top bar
        self.topbar = TopBar()
        main_layout.addWidget(self.topbar)
        # Middle layout
        middle_layout = QHBoxLayout()
        self.sidebar = Sidebar()
        self.device_panel = DevicePanel()
        middle_layout.addWidget(self.sidebar)
        middle_layout.addWidget(self.device_panel)
        main_layout.addLayout(middle_layout)
        # Bottom tools
        self.bottom_tools = BottomTools()
        main_layout.addWidget(self.bottom_tools)

from PyQt6.QtWidgets import QWidget, QHBoxLayout, QLabel, QPushButton
from PyQt6.QtCore import Qt
class TopBar(QWidget):
    def __init__(self):
        super().__init__()
        self.setFixedHeight(70)
        self.setStyleSheet("""
            QWidget {
                background: qlineargradient(
                    x1:0, y1:0, x2:1, y2:0,
                    stop:0 #1e6dfb, stop:1 #0b4fd1
                );
                color: white;
            }
        """)
        layout = QHBoxLayout(self)
        title = QLabel("eTool")
        title.setStyleSheet("font-size: 22px; font-weight: bold;")
        layout.addWidget(title)
        layout.addStretch()
        for name in ["iDevice", "Android", "Flash", "Toolbox"]:
            btn = QPushButton(name)
            btn.setStyleSheet("color:white; padding:8px;")
            layout.addWidget(btn)

from PyQt6.QtWidgets import QDialog, QVBoxLayout, QLabel, QPushButton
class ConnectionDialog(QDialog):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Connection Reminder")
        self.resize(400, 250)
        layout = QVBoxLayout(self)
        msg = QLabel("✅ Device is connected!\n\nCheck USB cable quality if unstable.")
        msg.setWordWrap(True)
        ok_btn = QPushButton("OK")
        ok_btn.clicked.connect(self.accept)
        layout.addWidget(msg)
        layout.addWidget(ok_btn)
